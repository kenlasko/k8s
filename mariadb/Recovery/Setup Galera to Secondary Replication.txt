flush tables with read lock;
show variables like 'gtid_binlog_pos';

# From another window on same cluster member:
mysqldump -h mariadb.mariadb.svc.cluster.local -u root -p***REMOVED*** -B homeassist phpmyadmin ucdialplans vaultwarden > /bitnami/mariadb/data/mariadb-repl-backup.sql

# Once done, then unlock from first:
unlock tables;

# From NUC1 host
# mount the Longhorn vol for MariaDB-standalone
sudo mount /dev/longhorn/pvc-f56c9d0a-d9cb-4b13-92ce-0828378f7b88 /mnt/
mv /kube-storage/mariadb/data/mariadb-repl-backup.sql /mnt/mariadb-repl-backup.sql

# Then on mariadb-standalone pod command prompt:
mariadb -u root -p***REMOVED***
stop slave;

drop database homeassist;
drop database ucdialplans;
drop database vaultwarden;

# Exit to prompt and run
mariadb -u root -p***REMOVED*** < /bitnami/mariadb/mariadb-repl-backup.sql

# Then run 
set global gtid_slave_pos = "0-1-1036026";

    master_host='mariadb.mariadb.svc.cluster.local',
    master_user='replicator',
    master_password='***REMOVED***',
    master_port=3306,
    master_connect_retry=10,
    master_use_gtid=slave_pos;
